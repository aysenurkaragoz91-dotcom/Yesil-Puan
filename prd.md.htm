    <script>

        const database = {

            "global fast fashion": { score: 25, cat: "Tekstil", alt: "Yerel Atölye X", reason: "Yüksek su tüketimi ve plastik ambalaj." },

            "endüstriyel süt": { score: 40, cat: "Gıda", alt: "Yeşil Mandıra", reason: "Karbon ayak izi yüksek dağıtım ağı." },

            "yerel atölye x": { score: 92, cat: "Tekstil", alt: null, reason: "Geri dönüştürülmüş kumaş ve adil ticaret." },

            "yeşil mandıra": { score: 95, cat: "Gıda", alt: null, reason: "Cam şişe ve 50km çapında yerel dağıtım." }

        };

 

        function searchBrand() {

            const input = document.getElementById('brandInput').value.toLowerCase();

            const resultArea = document.getElementById('resultArea');

            const data = database[input];

 

            if (data) {

                resultArea.classList.remove('hidden');

                document.getElementById('resBrandName').innerText = input;

                document.getElementById('resCategory').innerText = data.cat;

                document.getElementById('resScoreCircle').innerText = data.score;

               

                const scoreCard = document.getElementById('scoreCard');

                const alertBox = document.getElementById('alertBox');

                const scoreCircle = document.getElementById('resScoreCircle');

 

                if (data.score < 50) {

                    scoreCard.style.borderColor = "#ef4444";

                    scoreCircle.style.backgroundColor = "#ef4444";

                    alertBox.className = "mt-8 p-6 rounded-2xl flex items-start gap-4 bg-red-50 border border-red-100 text-red-800";

                    document.getElementById('alertIcon').className = "fa-solid fa-circle-exclamation text-2xl mt-1";

                    document.getElementById('alertTitle').innerText = "Dikkat: Düşük Puan!";

                    document.getElementById('alertDesc').innerText = `${data.reason} Bunun yerine ${data.alt} markasını tercih edebilirsin.`;

                    document.getElementById('switchBtn').classList.remove('hidden');

                } else {

                    scoreCard.style.borderColor = "#22c55e";

                    scoreCircle.style.backgroundColor = "#22c55e";

                    alertBox.className = "mt-8 p-6 rounded-2xl flex items-start gap-4 bg-green-50 border border-green-100 text-green-800";

                    document.getElementById('alertIcon').className = "fa-solid fa-circle-check text-2xl mt-1";

                    document.getElementById('alertTitle').innerText = "Harika Seçim!";

                    document.getElementById('alertDesc').innerText = data.reason;

                    document.getElementById('switchBtn').classList.add('hidden');

                }

               

                window.scrollTo({ top: resultArea.offsetTop - 100, behavior: 'smooth' });

            } else {

                alert("Marka veritabanında bulunamadı. Lütfen 'Global Fast Fashion' veya 'Yeşil Mandıra' yazarak test edin.");

            }

        }

    </script>
